    <script>
        const button = document.getElementById("mostrarFlores");
        button.addEventListener("click", function() {
            button.classList.add("clicked");
            crearCorazones();
            crearMensajes();
        });

        // Llamar a crearCorazones y crearMensajes automáticamente cada 3 segundos
        setInterval(() => {
            crearCorazones();
            crearMensajes();
        }, 3000);

        function crearCorazones() {
            const cantidadCorazones = 100;
            const contenedor = document.body;

            for (let i = 0; i < cantidadCorazones; i++) {
                const corazon = document.createElement("div");
                corazon.classList.add("corazon");
                corazon.style.left = Math.random() * window.innerWidth + "px";
                corazon.style.top = Math.random() * window.innerHeight + "px";
                contenedor.appendChild(corazon);

                setTimeout(() => {
                    corazon.remove();
                }, 5000);
            }
        }

        function crearMensajes() {
            const mensajes = ["Para Ale", "Loveuuuu", "Hermosa", "Vida"];
            const contenedor = document.body;

            mensajes.forEach((mensaje, index) => {
                const mensajeElement = document.createElement("div");
                mensajeElement.classList.add("mensaje");
                mensajeElement.innerText = mensaje;
                mensajeElement.style.left = Math.random() * window.innerWidth + "px";
                mensajeElement.style.top = Math.random() * window.innerHeight + "px";
                mensajeElement.style.animationDelay = `${index * 0.5}s`; // Agregar un retraso a la animación
                contenedor.appendChild(mensajeElement);

                setTimeout(() => {
                    mensajeElement.remove();
                }, 5000);
            });
        }
    </script>
